def destroy
  @game = @result.game
  @result.destroy

  respond_to do |format|
    format.html { redirect_to @game }
    format.json { head :no_content }
  end
end

def recall
  former_domminating = @result.dominating
  
  @result.recall!(params[:pool])
  
  if former_domminating == 'pain' and @result.dominating != 'pain'
    game = @result.game
    game.despair -= 1
    game.save
  end

  respond_to do |format|
    format.js
  end
end
